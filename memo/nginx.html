NGINX

Install

sudo apt install nginx

Start/Stop/Reload

  • Start: sudo nginx
  • Stop: sudo nginx -s stop
  • Reload: sudo nginx -s reload

Config

/etc/nginx/sites-avali/default

    server 127.0.0.1:8888;
}

server {
        listen 8080 default_server;
        listen [::]:8080 default_server;

        # SSL configuration
        #
        listen 443 ssl default_server;
        listen [::]:443 ssl default_server;

        ssl_certificate      /etc/nginx/ssl/server+inca.pem;
        ssl_certificate_key  /etc/nginx/ssl/server.key;

        #
        # Note: You should disable gzip for SSL traffic.
        # See: https://bugs.debian.org/773332
        #
        # Read up on ssl_ciphers to ensure a secure configuration.
        # See: https://bugs.debian.org/765782

        # Self signed certs generated by the ssl-cert package
        # Don't use them in a production server!

        root /var/www/html;

        # Add index.php to the list if you are using PHP
        index index.html index.htm index.nginx-debian.html;

        server_name _;

        location / {
                rewrite (.*) $1 break;
                proxy_pass http://127.0.0.1:8887/html$1;
        }

        location /app {
                rewrite /app(.*) $1 break;
                proxy_pass http://127.0.0.1:8888$1;
        }

        location /wss {
                proxy_pass http://127.0.0.1:8888;
                proxy_http_version 1.1;
                proxy_set_header Upgrade $http_upgrade;
                proxy_set_header Connection "upgrade";
		proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        }

        location /scratch {
                rewrite /scratch(.*) $1 break;
                proxy_pass http://127.0.0.1:8887/scratch$1;
        }

        location /nginx {
                # First attempt to serve request as file, then
                # as directory, then fall back to displaying a 404.
                try_files $uri $uri/ =404;
